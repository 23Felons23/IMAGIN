```python
"""
render_bridge.py — Translates pipeline outputs into Remotion-ready render_config.json
and triggers npx remotion render subprocess per highlight clip.
"""
import json
import os
import subprocess
from pathlib import Path


def build_render_config(
    highlight: dict,
    transcript_words: list[dict],
    video_files: dict[str, str],  # {speaker_label: video_file_path}
    emoji_map: dict[str, str],    # from emojis.ts mirrored here for Python use
) -> dict:
    """Build the renderConfig JSON object for a single highlight clip."""
    clip_words = [
        w for w in transcript_words
        if w["end"] >= highlight["start"] and w["start"] <= highlight["end"]
    ]
    # Normalize word timestamps relative to clip start
    offset = highlight["start"]
    for w in clip_words:
        w["start"] = round(w["start"] - offset, 3)
        w["end"] = round(w["end"] - offset, 3)

    return {
        "clipStart": highlight["start"],
        "clipEnd": highlight["end"],
        "durationSeconds": round(highlight["end"] - highlight["start"], 3),
        "videoFiles": video_files,
        "words": clip_words,
        "score": highlight.get("score", 0),
        "reason": highlight.get("reason", ""),
    }


def render_clip(
    render_config: dict,
    clip_index: int,
    output_dir: str,
    remotion_root: str,
) -> str:
    """
    Write render_config.json and trigger npx remotion render for one clip.
    Returns path to the output MP4 file.
    """
    output_path = Path(output_dir)
    output_path.mkdir(parents=True, exist_ok=True)

    config_path = output_path / f"render_config_{clip_index:02d}.json"
    with open(config_path, "w") as f:
        json.dump(render_config, f, indent=2)

    output_mp4 = output_path / f"clip_{clip_index:02d}.mp4"
    duration_frames = round(render_config["durationSeconds"] * 30)  # 30fps

    cmd = [
        "npx", "remotion", "render",
        "src/index.ts",           # Remotion entry point
        "PodcastClip",            # Composition ID
        str(output_mp4),
        f"--props={config_path}",
        f"--frames=0-{duration_frames}",
        "--codec=h264",
        "--pixel-format=yuv420p",
        "--log=verbose",
    ]
    result = subprocess.run(cmd, cwd=remotion_root, capture_output=True, text=True)
    if result.returncode != 0:
        raise RuntimeError(f"Remotion render failed for clip {clip_index}:\n{result.stderr}")

    return str(output_mp4)


DEFAULT_CAMERA_CONFIG = {}  # fallback: all speakers → camera1.mp4


def render_all_clips(
    job_id: str,
    tmp_dir: str,
    camera_config_path: str | None,
    remotion_root: str,
) -> list[str]:
    """
    Full render orchestration for a job:
    1. Load highlights, transcript, camera config
    2. Build render_config per clip
    3. Trigger remotion render per clip
    4. Return list of output MP4 paths
    """
    job_dir = Path(tmp_dir) / job_id
    clips_dir = job_dir / "clips"

    highlights = json.loads((job_dir / "highlights.json").read_text())
    transcript_words = json.loads((job_dir / "transcript.json").read_text())

    # Load camera config if provided, else use default
    if camera_config_path and Path(camera_config_path).exists():
        camera_config = json.loads(Path(camera_config_path).read_text())
    else:
        camera_config = DEFAULT_CAMERA_CONFIG

    # Build a video_files map: speaker → full path to video file
    video_files = {}
    for speaker, cam_filename in camera_config.items():
        cam_path = str(job_dir / cam_filename)
        video_files[speaker] = cam_path
    # Fallback: if no config provided, map everything to camera1.mp4
    if not video_files:
        fallback = str(job_dir / "camera1.mp4")
        video_files = {"DEFAULT": fallback}

    output_paths = []
    for i, highlight in enumerate(highlights, start=1):
        config = build_render_config(
            highlight=highlight,
            transcript_words=list(transcript_words),  # copy to avoid mutation
            video_files=video_files,
            emoji_map={},  # emoji matching done in Remotion TS layer
        )
        output_mp4 = render_clip(
            render_config=config,
            clip_index=i,
            output_dir=str(clips_dir),
            remotion_root=remotion_root,
        )
        output_paths.append(output_mp4)

    return output_paths
```
