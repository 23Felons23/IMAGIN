```python
import os

def get_llm_client():
    """Return a completion function based on LLM_PROVIDER env var."""
    provider = os.environ.get("LLM_PROVIDER", "openai").lower()
    if provider == "openai":
        from openai import OpenAI
        client = OpenAI(api_key=os.environ.get("OPENAI_API_KEY"))
        def complete(system_prompt: str, user_prompt: str) -> str:
            resp = client.chat.completions.create(
                model="gpt-4o",
                messages=[
                    {"role": "system", "content": system_prompt},
                    {"role": "user", "content": user_prompt}
                ],
                temperature=0.2,
            )
            return resp.choices[0].message.content
        return complete
    elif provider == "anthropic":
        import anthropic
        client = anthropic.Anthropic(api_key=os.environ.get("ANTHROPIC_API_KEY"))
        def complete(system_prompt: str, user_prompt: str) -> str:
            msg = client.messages.create(
                model="claude-haiku-20240307",
                max_tokens=1024,
                system=system_prompt,
                messages=[{"role": "user", "content": user_prompt}]
            )
            return msg.content[0].text
        return complete
    elif provider == "ollama":
        import requests
        base_url = os.environ.get("OLLAMA_BASE_URL", "http://localhost:11434")
        model = os.environ.get("OLLAMA_MODEL", "mistral")
        def complete(system_prompt: str, user_prompt: str) -> str:
            resp = requests.post(f"{base_url}/api/chat", json={
                "model": model,
                "messages": [
                    {"role": "system", "content": system_prompt},
                    {"role": "user", "content": user_prompt}
                ],
                "stream": False
            })
            return resp.json()["message"]["content"]
        return complete
    else:
        raise ValueError(f"Unknown LLM_PROVIDER: {provider}")
```

<action>
Create `backend/highlight_extractor.py`.

Implement the core sliding window logic and multimodal energy extractor:

1. `chunk_transcript(words, window_seconds=300, overlap_seconds=30)` — yields overlapping slices of the word list. Each chunk is a dict with `words`, `chunk_start` (float), `chunk_end` (float).

2. `score_audio_energy(audio_path, chunk_start, chunk_end)` — uses `librosa` to load the audio segment and compute mean RMS energy. Returns a float 0–1 (normalized by max RMS of the full file).

3. `score_chunk_with_llm(words_chunk, llm_complete)` — formats the chunk as readable transcript text with timestamps, sends to LLM with a system prompt asking it to:
   - Return a JSON object: `{"start": float, "end": float, "score": float (0-1), "reason": str}`
   - Pick the single most engaging CONTIGUOUS sub-segment within the chunk
   - Never pick a start/end outside the chunk's boundaries
   - Score 0–1 for overall engagement

4. `extract_highlights_multimodal(transcript_path, audio_path, top_n=3, window_seconds=300)` — orchestrates: chunk → score audio energy → LLM score → combined score (0.4 * audio_energy + 0.6 * llm_score) → return top N results sorted by combined score.
</action>

- Returns `list[dict]` with keys: `start`, `end`, `score`, `reason`
</acceptance_criteria>
